Q.1: Natural Join for Department Addresses

SQL:
SELECT EMPNO, ENAME, SAL, DNAME, LOC
FROM EMP NATURAL JOIN DEPT;

Q.2: Equi Join for Salesmen

SQL:
SELECT E.JOB, E.MGR, E.SAL, E.COMM, D.DNAME
FROM EMP E, DEPT D
WHERE E.DEPTNO = D.DEPTNO
AND E.JOB = 'SALESMAN';

Q.3: Employees working in DALLAS

SQL:

SELECT E.ENAME, E.JOB, E.DEPTNO, D.DNAME
FROM EMP E, DEPT D
WHERE E.DEPTNO = D.DEPTNO
AND D.LOC = 'DALLAS';

Q.4: Self Join for Employee and Manager Info

SQL:
SELECT E.ENAME AS "Employee", E.EMPNO AS "Emp#", M.ENAME AS "Manager", M.EMPNO AS "Mgr#"
FROM EMP E, EMP M
WHERE E.MGR = M.EMPNO;

Q.5: Outer Join to include King (No Manager)

SQL:
SELECT E.ENAME AS "Employee", E.EMPNO AS "Emp#", M.ENAME AS "Manager", M.EMPNO AS "Mgr#"
FROM EMP E LEFT OUTER JOIN EMP M
ON E.MGR = M.EMPNO
ORDER BY E.EMPNO;

Q.6: Non-Equi Join with SALGRADE

SQL :
To show the structure
DESC SALGRADE;
Query to fetch grades based on salary range
SELECT E.ENAME, E.JOB, D.DNAME, E.SAL, S.GRADE
FROM EMP E, DEPT D, SALGRADE S
WHERE E.DEPTNO = D.DEPTNO
AND E.SAL BETWEEN S.LOSAL AND S.HISAL;

Q.7: Display all Departments, even those without Employees

SQL:
SELECT E.ENAME, D.DNAME
FROM EMP E RIGHT OUTER JOIN DEPT D
ON E.DEPTNO = D.DEPTNO;

Q.8: Employees hired before their Managers

SQL:
SELECT E.ENAME AS "Emp Name", E.HIREDATE AS "Emp Hiredate",
M.ENAME AS "Mgr Name", M.HIREDATE AS "Mgr Hiredate"
FROM EMP E JOIN EMP M
ON E.MGR = M.EMPNO
WHERE E.HIREDATE < M.HIREDATE;

Q.9: Join using the USING clause for CLERKs

SQL:
SELECT EMPNO, ENAME, DNAME, LOC
FROM EMP JOIN DEPT USING (DEPTNO)
WHERE JOB = 'CLERK';

Q.10: Join using ON clause (Salary > 2000)

SQL :
SELECT E.ENAME, E.SAL, E.MGR, D.DNAME
FROM EMP E JOIN DEPT D
ON E.DEPTNO = D.DEPTNO
WHERE E.SAL > 2000;

Q.11: LEFT OUTER JOIN

SQL :
SELECT E.EMPNO, E.ENAME, E.JOB, E.DEPTNO, D.DNAME, D.LOC
FROM EMP E LEFT OUTER JOIN DEPT D
ON E.DEPTNO = D.DEPTNO;

Q.12: RIGHT OUTER JOIN

SQL :
SELECT E.ENAME, D.DNAME
FROM EMP E RIGHT OUTER JOIN DEPT D
ON E.DEPTNO = D.DEPTNO;

Q.13: FULL OUTER JOIN

SQL:
SELECT E.EMPNO, D.DNAME, D.LOC
FROM EMP E FULL OUTER JOIN DEPT D
ON E.DEPTNO = D.DEPTNO;
